### Task 3: Render the Today's bar

- [ ] **3.1** `app.js` `renderToday()`:
```js
function renderToday(){
  if(!DOM.todaySlots) return;
  DOM.todayDateLabel.textContent = new Date().toLocaleDateString(undefined,{weekday:'short',month:'short',day:'numeric'});
  const d = daySchedule();
  DOM.todaySlots.innerHTML = SLOT_TIMES.map(s => {
    const id = d[s]; const task = id ? state.tasks.find(t=>t.id===id) : null;
    if(id && !task){ delete d[s]; }                    // dangling (deleted task) -> clear
    const cat = task ? (state.categories.find(c=>c.id===task.category)||{color:'#64748b'}) : null;
    const chip = task
      ? `<div class="today-chip${task.status==='done'?' done':''}" draggable="true" data-id="${task.id}" style="--cat:${cat.color}">
           <span class="tc-dot"></span><span class="tc-title">${escapeHTML(task.title)}</span>
           <button class="tc-x" data-slot="${s}" aria-label="Unschedule"><i class="ri-close-line"></i></button>
         </div>`
      : '';
    return `<div class="today-slot${task?'':' empty'}" data-slot="${s}">
              <div class="today-time">${fmtSlot(s)}</div>
              <div class="today-drop">${chip}</div>
            </div>`;
  }).join('');
}
```
